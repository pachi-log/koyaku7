let activeTab = 'tab-header';

const STORAGE_KEY = 'pachidiff-dashboard-v2';

// 台ごとの固定カラー（視認性高め）
const UNIT_COLORS = {
  '1213':'#f0a500','1214':'#1a8cff','1215':'#00e8a2',
  '1216':'#ff3f6c','1217':'#c77dff','1218':'#7eb8ff',
  '1219':'#ffb347','1220':'#00d4aa','1221':'#ffe066',
  '1222':'#ff7eb3','1223':'#4dffb4'
};
const _colorCache = {};
function getColor(u) {
  if (UNIT_COLORS[u]) return UNIT_COLORS[u];
  if (!_colorCache[u]) {
    const h = [...u].reduce((a,c)=>a+c.charCodeAt(0),0);
    _colorCache[u] = `hsl(${(h*53)%360},75%,62%)`;
  }
  return _colorCache[u];
}

/* ══ STORAGE ══ */
async function saveToStorage() {
  const payload = JSON.stringify({
    title:state.title, sub:state.sub,
    units:state.units, data:state.data, currentUnit
  });
  let ok = false;
  try {
    const r = await window.storage.set(STORAGE_KEY, payload);
    if (r) ok = true;
  } catch(e) {}
  if (!ok) {
    try { localStorage.setItem(STORAGE_KEY, payload); ok = true; } catch(e) {}
  }
  showToast(ok ? '✓ 保存しました' : '⚠ 保存できませんでした', !ok);
}

async function loadFromStorage() {
  let saved = null;
  try {
    const r = await window.storage.get(STORAGE_KEY);
    if (r && r.value) saved = JSON.parse(r.value);
  } catch(e) {}
  if (!saved) {
    try { const ls = localStorage.getItem(STORAGE_KEY); if(ls) saved = JSON.parse(ls); } catch(e) {}
  }
  if (saved) {
    if (saved.title) state.title = saved.title;
    if (saved.sub !== undefined) state.sub = saved.sub;
    if (saved.units && saved.units.length) state.units = saved.units;
    if (saved.data) state.data = saved.data;
    if (saved.currentUnit && saved.units && saved.units.includes(saved.currentUnit))
      currentUnit = saved.currentUnit;
    document.getElementById('displayTitle').textContent = state.title;
    document.getElementById('displaySub').textContent = state.sub;
  }
}

function showToast(msg, isErr=false) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.style.background = isErr
    ? 'rgba(255,63,108,0.15)' : 'rgba(0,232,162,0.12)';
  t.style.border = isErr
    ? '1px solid rgba(255,63,108,0.4)' : '1px solid rgba(0,232,162,0.35)';
  t.style.color = isErr ? '#ff3f6c' : '#00e8a2';
  t.style.boxShadow = isErr
    ? '0 8px 24px rgba(255,63,108,0.15)' : '0 8px 24px rgba(0,232,162,0.15)';
  t.style.opacity = '1';
  t.style.transform = 'translateX(-50%) translateY(0)';
  clearTimeout(t._t);
  t._t = setTimeout(() => {
    t.style.opacity = '0';
    t.style.transform = 'translateX(-50%) translateY(12px)';
  }, 2000);
}

/* ══ HELPERS ══ */
function calcCum(rows) {
  let c = 0;
  return rows.map(r => { c += r.diff; return { date: r.date, diff: r.diff, cum: c }; });
}
function getFinal(u) {
  const d = state.data[u];
  if (!d || !d.length) return 0;
  const r = calcCum(d); return r[r.length-1].cum;
}
function getToday(u) {
  const d = state.data[u];
  if (!d || !d.length) return 0;
  return d[d.length-1].diff;
}
function getMax(u) { return Math.max(...calcCum(state.data[u]).map(r=>r.cum)); }
function getMin(u) { return Math.min(...calcCum(state.data[u]).map(r=>r.cum)); }
function fmt(n) { return (n>=0?'+':'')+n.toLocaleString(); }

/* ══ UNIT SELECTOR ══ */
function buildUnitSelector() {
  const sel = document.getElementById('unitSelector');
  sel.innerHTML = '';
  state.units.forEach(u => {
    if (!state.data[u]) return;
    const f = getFinal(u);
    const btn = document.createElement('button');
    btn.className = 'unit-btn ' + (f>=0?'pos-unit':'neg-unit') + (u===currentUnit?' active':'');
    btn.textContent = u;
    btn.onclick = () => switchUnit(u);
    sel.appendChild(btn);
  });
}
function switchUnit(u) {
  if (!state.data[u]) return;
  currentUnit = u;
  buildUnitSelector();
  renderContent();
}

/* ══ MAIN RENDER ══ */
function renderContent() {
  if (!state.data[currentUnit]) {
    currentUnit = state.units.find(u=>state.data[u]) || '';
    if (!currentUnit) return;
  }
  const rawData = state.data[currentUnit];
  const data = calcCum(rawData);
  const final = getFinal(currentUnit);
  const maxV = getMax(currentUnit);
  const minV = getMin(currentUnit);
  const color = getColor(currentUnit);
  const isPos = final >= 0;
  const ranked = [...state.units].filter(u=>state.data[u]).sort((a,b)=>getFinal(b)-getFinal(a));
  const rank = ranked.indexOf(currentUnit) + 1;
  const maxAbs = Math.max(...ranked.map(u=>Math.abs(getFinal(u))), 1);
  const MEDALS = ['🥇','🥈','🥉'];

  // Ranking HTML
  const rankHTML = ranked.map((u, i) => {
    const f = getFinal(u);
    const td = getToday(u);
    const bw = Math.round(Math.abs(f)/maxAbs*100);
    const bc = f>=0 ? '#00e8a2' : '#ff3f6c';
    const isCur = u === currentUnit;
    const posEl = i < 3
      ? `<span class="rank-pos r${i+1}">${MEDALS[i]}</span>`
      : `<span class="rank-pos">${i+1}</span>`;
    return `<div class="rank-item${isCur?' current':''}">
      ${posEl}
      <span class="rank-unit-name" style="color:${getColor(u)}" onclick="switchUnit('${u}')">${u}</span>
      <div class="rank-bar-bg"><div class="rank-bar-fill" style="width:${bw}%;background:${bc}"></div></div>
      <span class="rank-today" style="color:${td>=0?'rgba(0,232,162,0.6)':'rgba(255,63,108,0.6)'}">${fmt(td)}</span>
      <span class="rank-val" style="color:${bc}">${fmt(f)}</span>
    </div>`;
  }).join('');

  // Table HTML
  let tableHTML = '';
  let lastMonth = '';
  data.forEach(r => {
    const month = r.date.split('/')[0];
    if (month !== lastMonth) {
      tableHTML += `<tr class="tr-month-sep"><td colspan="3">${month}月</td></tr>`;
      lastMonth = month;
    }
    const dc = r.diff >= 0 ? 'td-pos' : 'td-neg';
    const cc = r.cum  >= 0 ? 'td-cpos' : 'td-cneg';
    tableHTML += `<tr>
      <td class="td-date">${r.date}</td>
      <td class="td-diff ${dc}">${fmt(r.diff)}</td>
      <td class="td-cum  ${cc}">${fmt(r.cum)}</td>
    </tr>`;
  });

  document.getElementById('content').innerHTML = `
    <div class="unit-header">
      <div class="unit-num" style="color:${color};text-shadow:0 0 28px ${color}55">${currentUnit}</div>
      <div class="unit-meta">
        <div class="unit-badge" style="color:${color};border-color:${color}55">
          ${isPos?'PROFIT':'DEFICIT'}
        </div>
        <div class="unit-rank" style="color:${color}99">RANK  #${rank} / ${ranked.length}</div>
      </div>
    </div>

    <div class="stats-row">
      <div class="stat-card card-final">
        <div class="stat-label">最終差枚</div>
        <div class="stat-value ${isPos?'pos':'neg'}" id="anim-final">+0</div>
      </div>
      <div class="stat-card card-max">
        <div class="stat-label">最高到達</div>
        <div class="stat-value pos">${fmt(maxV)}</div>
      </div>
      <div class="stat-card card-min">
        <div class="stat-label">最低到達</div>
        <div class="stat-value neg">${fmt(minV)}</div>
      </div>
    </div>

    <div class="chart-card">
      <div class="chart-label">CUMULATIVE DIFF</div>
      <div class="chart-label-right" style="color:${color}99">${currentUnit}</div>
      <canvas id="mainChart" height="190"></canvas>
    </div>

    <div class="card">
      <div class="card-header">
        <span class="card-header-label">全台ランキング</span>
        <span class="card-header-label" style="color:var(--muted)">今日差枚 ／ 累積差枚</span>
      </div>
      ${rankHTML}
    </div>

    <div class="card">
      <div class="card-header">
        <span class="card-header-label">日別データ</span>
        <button class="card-header-action" onclick="openModal('data')">✏ データ編集</button>
      </div>
      <table>
        <thead><tr><th>日付</th><th>当日差枚</th><th>累積差枚</th></tr></thead>
        <tbody>${tableHTML}</tbody>
      </table>
    </div>`;

  // アニメーション
  animCount('anim-final', 0, final, 900);
  requestAnimationFrame(() => drawChart(data, color));
}

/* ══ COUNT UP ══ */
function animCount(id, from, to, dur) {
  const el = document.getElementById(id);
  if (!el) return;
  const s = performance.now();
  const step = now => {
    const t = Math.min((now-s)/dur, 1);
    const e = 1 - Math.pow(1-t, 3);
    el.textContent = fmt(Math.round(from + (to-from)*e));
    if (t < 1) requestAnimationFrame(step);
    else el.textContent = fmt(to);
  };
  requestAnimationFrame(step);
}

/* ══ CHART ══ */
function drawChart(data, color) {
  const canvas = document.getElementById('mainChart');
  if (!canvas) return;
  const dpr = Math.min(window.devicePixelRatio || 1, 2);
  const wrap = canvas.parentElement;
  const W = wrap.clientWidth - 0;
  const H = 190;
  canvas.width = W * dpr;
  canvas.height = H * dpr;
  canvas.style.width = W + 'px';
  canvas.style.height = H + 'px';

  const ctx = canvas.getContext('2d');
  ctx.scale(dpr, dpr);

  const values = data.map(r => r.cum);
  const maxV = Math.max(...values, 0);
  const minV = Math.min(...values, 0);
  const range = maxV - minV || 1;

  const pL=48, pR=10, pT=14, pB=44;
  const cW = W - pL - pR;
  const cH = H - pT - pB;

  const toX = i => pL + (i / Math.max(values.length-1, 1)) * cW;
  const toY = v => pT + (1 - (v - minV) / range) * cH;

  // Grid lines
  const step = range > 50000 ? 10000 : range > 20000 ? 5000 : range > 8000 ? 2000 : 1000;
  const gMin = Math.ceil(minV/step)*step;
  const gMax = Math.floor(maxV/step)*step;

  ctx.font = `${8 * Math.min(W/320, 1)}px "Space Mono", monospace`;
  ctx.textAlign = 'right';

  for (let g = gMin; g <= gMax; g += step) {
    const y = toY(g);
    ctx.strokeStyle = g === 0 ? 'rgba(255,255,255,0.15)' : 'rgba(255,255,255,0.04)';
    ctx.lineWidth = g === 0 ? 1.2 : 0.8;
    ctx.setLineDash(g === 0 ? [] : [3, 4]);
    ctx.beginPath(); ctx.moveTo(pL, y); ctx.lineTo(W-pR, y); ctx.stroke();
    ctx.setLineDash([]);
    const label = (g >= 0 ? '+' : '') + (g/1000) + 'k';
    ctx.fillStyle = g === 0 ? 'rgba(255,255,255,0.45)' : 'rgba(90,116,144,0.9)';
    ctx.fillText(label, pL-4, y+3);
  }

  const zY = toY(0);

  // Parse color to rgb
  let cr=0,cg=200,cb=120;
  if (color.startsWith('#') && color.length>=7) {
    cr=parseInt(color.slice(1,3),16);
    cg=parseInt(color.slice(3,5),16);
    cb=parseInt(color.slice(5,7),16);
  }

  // Area fills
  const drawArea = () => {
    // Positive area
    const gPos = ctx.createLinearGradient(0, pT, 0, pT+cH);
    gPos.addColorStop(0, `rgba(${cr},${cg},${cb},0.22)`);
    gPos.addColorStop(1, `rgba(${cr},${cg},${cb},0.02)`);
    ctx.beginPath();
    ctx.moveTo(toX(0), zY);
    values.forEach((v,i) => ctx.lineTo(toX(i), toY(v)));
    ctx.lineTo(toX(values.length-1), zY);
    ctx.closePath();
    ctx.fillStyle = gPos; ctx.fill();

    // Negative area
    ctx.save();
    ctx.beginPath();
    ctx.rect(pL, zY, cW+pR+2, cH+pB);
    ctx.clip();
    const gNeg = ctx.createLinearGradient(0, zY, 0, pT+cH);
    gNeg.addColorStop(0, 'rgba(255,63,108,0.02)');
    gNeg.addColorStop(1, 'rgba(255,63,108,0.2)');
    ctx.beginPath();
    ctx.moveTo(toX(0), zY);
    values.forEach((v,i) => ctx.lineTo(toX(i), toY(v)));
    ctx.lineTo(toX(values.length-1), zY);
    ctx.closePath();
    ctx.fillStyle = gNeg; ctx.fill();
    ctx.restore();
  };

  drawArea();

  // Animate line draw
  let prog = 0;
  const animate = () => {
    ctx.clearRect(pL-1, pT-2, cW+pR+4, cH+12);
    drawArea();
    const pts = Math.min(Math.ceil(prog * values.length), values.length);
    if (pts >= 2) {
      ctx.beginPath();
      for (let i=0; i<pts; i++) {
        if (i===0) ctx.moveTo(toX(i), toY(values[i]));
        else ctx.lineTo(toX(i), toY(values[i]));
      }
      ctx.strokeStyle = color;
      ctx.lineWidth = 2;
      ctx.lineJoin = 'round';
      ctx.shadowColor = color;
      ctx.shadowBlur = 8;
      ctx.stroke();
      ctx.shadowBlur = 0;

      if (pts === values.length) {
        const lx=toX(pts-1), ly=toY(values[pts-1]);
        ctx.beginPath(); ctx.arc(lx, ly, 4, 0, Math.PI*2);
        ctx.fillStyle = '#fff';
        ctx.shadowColor = color; ctx.shadowBlur = 10;
        ctx.fill(); ctx.shadowBlur = 0;
        pingAnim(ctx, lx, ly, cr, cg, cb);
        return;
      }
    }
    prog += 0.045; requestAnimationFrame(animate);
  };
  animate();

  // X-axis date labels
  const every = Math.ceil(values.length / (W < 380 ? 6 : 10));
  data.forEach((d, i) => {
    if (i % every === 0 || i === data.length-1) {
      ctx.save();
      ctx.fillStyle = 'rgba(90,116,144,0.85)';
      ctx.font = `${7 * Math.min(W/320,1)}px sans-serif`;
      ctx.textAlign = 'right';
      ctx.translate(toX(i), pT+cH+6);
      ctx.rotate(-Math.PI/4);
      ctx.fillText(d.date, 0, 0);
      ctx.restore();
    }
  });
}

function pingAnim(ctx, x, y, r, g, b) {
  let radius = 5, alpha = 0.9;
  const step = () => {
    ctx.beginPath(); ctx.arc(x, y, radius, 0, Math.PI*2);
    ctx.strokeStyle = `rgba(${r},${g},${b},${alpha})`;
    ctx.lineWidth = 1.2; ctx.stroke();
    radius += 0.7; alpha -= 0.1;
    if (alpha > 0) requestAnimationFrame(step);
  };
  step();
}

/* ══ MODAL ══ */
function openModal(tab) {
  document.getElementById('editTitle').value = state.title;
  document.getElementById('editSub').value = state.sub;
  buildUnitListEditor();
  buildUnitDataSelector();
  if (tab === 'data') {
    switchTab('tab-data');
    selectDataUnit(currentUnit);
  } else {
    switchTab('tab-header');
  }
  document.getElementById('modalOverlay').classList.add('open');
  document.body.style.overflow = 'hidden';
}
function closeModal() {
  document.getElementById('modalOverlay').classList.remove('open');
  document.body.style.overflow = '';
}

function switchTab(tabId) {
  activeTab = tabId;
  const tabs = ['tab-header','tab-units','tab-data'];
  document.querySelectorAll('.tab-btn').forEach((b,i) => b.classList.toggle('active', tabs[i]===tabId));
  document.querySelectorAll('.tab-panel').forEach(p => p.classList.toggle('active', p.id===tabId));
}

function buildUnitDataSelector() {
  const sel = document.getElementById('unitDataSelector');
  sel.innerHTML = '';
  state.units.forEach(u => {
    if (!state.data[u]) return;
    const chip = document.createElement('button');
    chip.className = 'data-chip' + (u===editingDataUnit?' active':'');
    chip.textContent = u;
    chip.id = 'chip-' + u;
    chip.onclick = () => selectDataUnit(u);
    sel.appendChild(chip);
  });
}
function selectDataUnit(u) {
  editingDataUnit = u;
  document.querySelectorAll('.data-chip').forEach(c=>c.classList.remove('active'));
  const chip = document.getElementById('chip-'+u);
  if (chip) chip.classList.add('active');
  document.getElementById('dataTextarea').value =
    (state.data[u]||[]).map(r=>r.date+','+r.diff).join('\n');
  document.getElementById('parseError').style.display = 'none';
}

function saveCurrentDataEdit() {
  const errEl = document.getElementById('parseError');
  const lines = document.getElementById('dataTextarea').value
    .split('\n').filter(l=>l.trim());
  const rows = [];
  for (let i=0; i<lines.length; i++) {
    const line = lines[i].trim();
    if (!line) continue;
    const parts = line.split(',');
    if (parts.length < 2) {
      errEl.textContent = `⚠ 行${i+1}: カンマが必要です → "${line}"`;
      errEl.style.display = 'block'; return false;
    }
    const date = parts[0].trim();
    const diff = parseInt(parts[1].trim().replace(/[^0-9\-]/g,''), 10);
    if (!date) { errEl.textContent = `⚠ 行${i+1}: 日付が空です`; errEl.style.display='block'; return false; }
    if (isNaN(diff)) { errEl.textContent = `⚠ 行${i+1}: 数値エラー`; errEl.style.display='block'; return false; }
    rows.push({ date, diff });
  }
  // 日付を古い順にソート（M/D 形式を数値比較）
  rows.sort((a, b) => {
    const toNum = d => {
      const [m, day] = d.split('/').map(Number);
      return m * 100 + (day || 0);
    };
    return toNum(a.date) - toNum(b.date);
  });
  errEl.style.display = 'none';
  state.data[editingDataUnit] = rows;
  return true;
}

function applyChanges() {
  if (activeTab === 'tab-data' && editingDataUnit) {
    if (!saveCurrentDataEdit()) return;
  }
  state.title = document.getElementById('editTitle').value.trim() || state.title;
  state.sub   = document.getElementById('editSub').value.trim();
  document.getElementById('displayTitle').textContent = state.title;
  document.getElementById('displaySub').textContent   = state.sub;

  const items = document.querySelectorAll('.unit-list-item');
  const newUnits=[], newData={};
  items.forEach(item => {
    const orig = item.dataset.origUnit;
    const name = item.querySelector('.unit-name-input').value.trim();
    if (!name) return;
    newUnits.push(name);
    newData[name] = orig && state.data[orig] ? state.data[orig] : (state.data[name]||[]);
  });
  state.units = newUnits;
  state.data  = newData;
  if (!state.units.includes(currentUnit) || !state.data[currentUnit]) {
    currentUnit = state.units.find(u=>state.data[u]&&state.data[u].length) || state.units[0] || '';
  }
  closeModal();
  buildUnitSelector();
  renderContent();
  saveToStorage();
}
